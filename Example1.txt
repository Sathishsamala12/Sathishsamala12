namespace ATI_OESAPI.REPOSITORY
{
    public class IdentificationMarkRepository: IIdentificationMarksRepository
    {

        ATI_OESContext aTI_OESContext = new ATI_OESContext();
        public int AddIdentificationMarks(IdentificationMark identification)
        {
            var data = aTI_OESContext.IdentificationMarks.Add(identification);
            aTI_OESContext.SaveChanges();
            return 1;
        }
        public string UpdateIdentificationMarks(IdentificationMark identificationMark)
        {
            var data = aTI_OESContext.IdentificationMarks.Where(a => a.IdentificationMarkId == identificationMark.IdentificationMarkId).FirstOrDefault();
            data.IdentificationMark1 = identificationMark.IdentificationMark1;
            data.IdentificationMarkId = identificationMark.IdentificationMarkId;
            aTI_OESContext.SaveChanges();
            return "Data Updated";
        }
        public List<IdentificationMark> GetIdentificationMarks()
        {
            List<IdentificationMark> data = aTI_OESContext.IdentificationMarks.ToList();
            return data;
        }
        public IdentificationMark GetIdentificationMarkById(int identificationMarkId)
        {
            var data = aTI_OESContext.IdentificationMarks.Where(a => a.IdentificationMarkId== identificationMarkId).FirstOrDefault();
            return data;
        }
        public bool DeleteIdentificationMarks(int identificationMarkId)
        {
            var data = aTI_OESContext.IdentificationMarks.Where(a => a.IdentificationMarkId == identificationMarkId).FirstOrDefault();
            aTI_OESContext.Remove(data);
            aTI_OESContext.SaveChanges();
            return true;
        }
    }
